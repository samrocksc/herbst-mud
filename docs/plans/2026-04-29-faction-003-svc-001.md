### Task 1: Create faction_routes.go with CRUD endpoints

**File:** `server/routes/faction_routes.go` (new)

Endpoints:
- `GET /api/factions` — list all factions with their category
- `GET /api/factions/:id` — get single faction with members
- `POST /api/factions` — create faction
- `PUT /api/factions/:id` — update faction
- `DELETE /api/factions/:id` — delete faction
- `GET /api/faction-categories` — list all categories
- `POST /api/faction-categories` — create category
- `GET /api/factions/:id/members` — list characters in a faction

Register in `server/main.go` under a `faction` group with admin auth middleware.

Use ent's `client.Faction.Query()` with eager loading of edges (`WithCategory`, `WithRequiredTags`, `WithCharacterFactions`, `WithSkills`).

**Step 1: Write the route file**

```go
package routes

import (
	"net/http"
	"strconv"

	"herbst-server/db"
	"herbst-server/db/faction"
	"herbst-server/db/factioncategory"
	"herbst-server/db/faction/characterfaction"

	"github.com/gin-gonic/gin"

func RegisterFactionRoutes(r *gin.Engine, client *db.Client) {
	f := r.Group("/api")
	f.Use(adminAuthMiddleware()) // reuse existing admin auth

	f.GET("/factions", listFactions(client))
	f.GET("/factions/:id", getFaction(client))
	f.POST("/factions", createFaction(client))
	f.PUT("/factions/:id", updateFaction(client))
	f.DELETE("/factions/:id", deleteFaction(client))
	f.GET("/factions/:id/members", getFactionMembers(client))

	f.GET("/faction-categories", listFactionCategories(client))
	f.POST("/faction-categories", createFactionCategory(client))
}

func listFactions(client *db.Client) gin.HandlerFunc {
	return func(c *gin.Context) {
		factions, err := client.Faction.Query().
			WithCategory().
			WithRequiredTags().
			All(c.Request.Context())
		if err != nil {
			c.JSON(http.StatusInternalServerError, gin.H{"error": err.Error()})
			return
		}
		// Transform to JSON-friendly format
		result := make([]gin.H, len(factions))
		for i, f := range factions {
			result[i] = factionToJSON(f)
		}
		c.JSON(http.StatusOK, result)
	}
}

func getFaction(client *db.Client) gin.HandlerFunc {
	return func(c *gin.Context) {
		id, err := strconv.Atoi(c.Param("id"))
		if err != nil {
			c.JSON(http.StatusBadRequest, gin.H{"error": "invalid id"})
			return
		}
		f, err := client.Faction.Query().
			Where(faction.ID(id)).
			WithCategory().
			WithRequiredTags().
			WithSkills().
			Only(c.Request.Context())
		if err != nil {
			c.JSON(http.StatusNotFound, gin.H{"error": "faction not found"})
			return
		}
		c.JSON(http.StatusOK, factionToJSON(f))
	}
}

func createFaction(client *db.Client) gin.HandlerFunc {
	return func(c *gin.Context) {
		var req struct {
			Name        string `json:"name" binding:"required"`
			DisplayName string `json:"display_name" binding:"required"`
			Description string `json:"description"`
			CategoryID  int    `json:"category_id"`
		}
		if err := c.ShouldBindJSON(&req); err != nil {
			c.JSON(http.StatusBadRequest, gin.H{"error": err.Error()})
			return
		}
		created, err := client.Faction.Create().
			SetName(req.Name).
			SetDisplayName(req.DisplayName).
			SetDescription(req.Description).
			SetCategoryID(req.CategoryID).
			Save(c.Request.Context())
		if err != nil {
			c.JSON(http.StatusInternalServerError, gin.H{"error": err.Error()})
			return
		}
		c.JSON(http.StatusCreated, factionToJSON(created))
	}
}

func updateFaction(client *db.Client) gin.HandlerFunc {
	return func(c *gin.Context) {
		id, err := strconv.Atoi(c.Param("id"))
		if err != nil {
			c.JSON(http.StatusBadRequest, gin.H{"error": "invalid id"})
			return
		}
		var req struct {
			DisplayName string `json:"display_name"`
			Description string `json:"description"`
			CategoryID  int    `json:"category_id"`
		}
		if err := c.ShouldBindJSON(&req); err != nil {
			c.JSON(http.StatusBadRequest, gin.H{"error": err.Error()})
			return
		}
		updated, err := client.Faction.UpdateOneID(id).
			SetDisplayName(req.DisplayName).
			SetDescription(req.Description).
			SetCategoryID(req.CategoryID).
			Save(c.Request.Context())
		if err != nil {
			c.JSON(http.StatusInternalServerError, gin.H{"error": err.Error()})
			return
		}
		c.JSON(http.StatusOK, factionToJSON(updated))
	}
}

func deleteFaction(client *db.Client) gin.HandlerFunc {
	return func(c *gin.Context) {
		id, err := strconv.Atoi(c.Param("id"))
		if err != nil {
			c.JSON(http.StatusBadRequest, gin.H{"error": "invalid id"})
			return
		}
		err = client.Faction.DeleteOneID(id, c.Request.Context())
		if err != nil {
			c.JSON(http.StatusInternalServerError, gin.H{"error": err.Error()})
			return
		}
		c.JSON(http.StatusNoContent, nil)
	}
}

func getFactionMembers(client *db.Client) gin.HandlerFunc {
	return func(c *gin.Context) {
		id, err := strconv.Atoi(c.Param("id"))
		if err != nil {
			c.JSON(http.StatusBadRequest, gin.H{"error": "invalid id"})
			return
		}
		members, err := client.CharacterFaction.Query().
			Where(characterfaction.HasFactionWith(faction.ID(id))).
			WithCharacter().
			All(c.Request.Context())
		if err != nil {
			c.JSON(http.StatusInternalServerError, gin.H{"error": err.Error()})
			return
		}
		result := make([]gin.H, len(members))
		for i, m := range members {
			result[i] = gin.H{
				"character_id": m.Characters[0].ID,
				"name":        m.Characters[0].Name,
				"reputation":  m.Reputation,
				"status":      m.Status,
				"joined_at":   m.JoinedAt,
			}
		}
		c.JSON(http.StatusOK, result)
	}
}

func listFactionCategories(client *db.Client) gin.HandlerFunc {
	return func(c *gin.Context) {
		cats, err := client.FactionCategory.Query().All(c.Request.Context())
		if err != nil {
			c.JSON(http.StatusInternalServerError, gin.H{"error": err.Error()})
			return
		}
		result := make([]gin.H, len(cats))
		for i, cat := range cats {
			result[i] = gin.H{
				"id":             cat.ID,
				"name":           cat.Name,
				"display_name":   cat.DisplayName,
				"description":    cat.Description,
				"max_memberships": cat.MaxMemberships,
				"auto_join":      cat.AutoJoin,
			}
		}
		c.JSON(http.StatusOK, result)
	}
}

func createFactionCategory(client *db.Client) gin.HandlerFunc {
	return func(c *gin.Context) {
		var req struct {
			Name           string `json:"name" binding:"required"`
			DisplayName    string `json:"display_name" binding:"required"`
			Description    string `json:"description"`
			MaxMemberships int    `json:"max_memberships"`
			AutoJoin       bool   `json:"auto_join"`
		}
		if err := c.ShouldBindJSON(&req); err != nil {
			c.JSON(http.StatusBadRequest, gin.H{"error": err.Error()})
			return
		}
		created, err := client.FactionCategory.Create().
			SetName(req.Name).
			SetDisplayName(req.DisplayName).
			SetDescription(req.Description).
			SetMaxMemberships(req.MaxMemberships).
			SetAutoJoin(req.AutoJoin).
			Save(c.Request.Context())
		if err != nil {
			c.JSON(http.StatusInternalServerError, gin.H{"error": err.Error()})
			return
		}
		c.JSON(http.StatusCreated, gin.H{
			"id":              created.ID,
			"name":            created.Name,
			"display_name":    created.DisplayName,
			"description":    created.Description,
			"max_memberships": created.MaxMemberships,
			"auto_join":       created.AutoJoin,
		})
	}
}

func factionToJSON(f *db.Faction) gin.H {
	result := gin.H{
		"id":           f.ID,
		"name":         f.Name,
		"display_name": f.DisplayName,
		"description":  f.Description,
	}
	if f.Edges.Category != nil {
		result["category"] = gin.H{
			"id":           f.Edges.Category.ID,
			"name":         f.Edges.Category.Name,
			"display_name": f.Edges.Category.DisplayName,
		}
	}
	if f.Edges.RequiredTags != nil {
		tags := make([]string, len(f.Edges.RequiredTags))
		for i, t := range f.Edges.RequiredTags {
			tags[i] = t.RequiredTag
		}
		result["required_tags"] = tags
	}
	return result
}
```

```go
package services

import (
	"context"
	"errors"

	"herbst-server/db"
	"herbst-server/db/character"
)

// CharacterService handles character business logic
type CharacterService struct {
	client *db.Client
}

// NewCharacterService creates a new character service
func NewCharacterService(client *db.Client) *CharacterService {
	return &CharacterService{client: client}
}

// ErrCharacterNotFound is returned when a character doesn't exist
var ErrCharacterNotFound = errors.New("character not found")

// CreateCharacterInput is the input for creating a character
type CreateCharacterInput struct {
	UserID    int
	Name      string
	Password  string
	Class     string
	Race      string
	Gender    string
	RoomID    int
}

// CreateCharacter creates a new player character
func (s *CharacterService) CreateCharacter(ctx context.Context, input CreateCharacterInput) (*db.Character, error) {
	// Check name uniqueness
	existing, err := s.client.Character.Query().
		Where(character.NameEQ(input.Name)).
		Only(ctx)
	if err == nil && existing != nil {
		return nil, errors.New("character name already exists")
	}

	// Get class config (GameConfig lookup)
	classConfig := getClassConfig(input.Class)

	// Build character
	builder := s.client.Character.Create().
		SetName(input.Name).
		SetPassword(input.Password). // hash in route handler before calling service
		SetIsNPC(false).
		SetCurrentRoomId(input.RoomID).
		SetStartingRoomId(input.RoomID).
		SetHitpoints(classConfig.BaseHP).
		SetMaxHitpoints(classConfig.BaseHP).
		SetStamina(classConfig.BaseStamina).
		SetMaxStamina(classConfig.BaseStamina).
		SetMana(classConfig.BaseMana).
		SetMaxMana(classConfig.BaseMana).
		SetRace(input.Race).
		SetClass(input.Class).
		SetGender(input.Gender).
		SetLevel(1).
		SetXp(0).
		SetConstitution(classConfig.BaseConstitution).
		SetStrength(classConfig.BaseStrength).
		SetDexterity(classConfig.BaseDexterity).
		SetIntelligence(classConfig.BaseIntelligence).
		SetWisdom(classConfig.BaseWisdom).
		SetSkillBlades(classConfig.StartingSkills["blades"]).
		SetSkillStaves(classConfig.StartingSkills["staves"]).
		SetSkillKnives(classConfig.StartingSkills["knives"]).
		SetSkillMartial(classConfig.StartingSkills["martial"]).
		SetSkillBrawling(classConfig.StartingSkills["brawling"]).
		SetSkillTech(classConfig.StartingSkills["tech"]).
		SetSkillLightArmor(classConfig.StartingSkills["light_armor"]).
		SetSkillClothArmor(classConfig.StartingSkills["cloth_armor"]).
		SetSkillHeavyArmor(classConfig.StartingSkills["heavy_armor"])

	char, err := builder.Save(ctx)
	if err != nil {
		return nil, err
	}

	// Grant starting skills
	for skill, level := range classConfig.StartingSkills {
		if level == 0 {
			continue
		}
		skillID := classConfig.SkillIDs[skill]
		if skillID == "" {
			continue
		}
		_, err := s.client.CharacterSkill.Create().
			SetCharacterID(char.ID).
			SetSkillID(skillID).
			SetLevel(level).
			Save(ctx)
		if err != nil {
			// Log but don't fail
		}
	}

	return char, nil
}

// DeleteCharacter deletes a character and its related data
func (s *CharacterService) DeleteCharacter(ctx context.Context, charID int) error {
	char, err := s.client.Character.Get(ctx, charID)
	if err != nil {
		return ErrCharacterNotFound
	}

	// Delete related records
	_, _ = s.client.CharacterSkill.Delete().
		Where(). // TODO: add proper where clause
		Exec(ctx)

	return s.client.Character.DeleteOne(char, ctx)
}
```

```go
	"herbst-mud/admin-tui/style"
)

type FactionsModel struct {
	factions   []api.Faction
	categories []api.FactionCategory
	loading    bool
	errMsg     string
	selected   int
	width      int
	height     int
}

func NewFactionsScreen(token string) tea.Model {
	m := FactionsModel{loading: true}
	return m
}

func (m FactionsModel) Init() tea.Cmd {
	return func() tea.Msg {
		factions, err := api.ListFactions()
		if err != nil {
			return FactionsErrMsg{Err: err}
		}
		categories, err := api.ListFactionCategories()
		if err != nil {
			return FactionsErrMsg{Err: err}
		}
		return FactionsMsg{Factions: factions, Categories: categories}
	}
}

func (m FactionsModel) Update(msg tea.Msg) (tea.Model, tea.Cmd) {
	switch msg := msg.(type) {
	case tea.WindowSizeMsg:
		m.width = msg.Width
		m.height = msg.Height
	case FactionsMsg:
		m.factions = msg.Factions
		m.categories = msg.Categories
		m.loading = false
	case FactionsErrMsg:
		m.errMsg = fmt.Sprintf("Failed to load factions: %v", msg.Err)
		m.loading = false
	case tea.KeyMsg:
		switch msg.String() {
		case "ctrl+c":
			return m, tea.Quit
		case "up", "k":
			if m.selected > 0 {
				m.selected--
			}
		case "down", "j":
			if m.selected < len(m.factions)-1 {
				m.selected++
			}
		}
	}
	return m, nil
}

func (m FactionsModel) View() string {

}
```

```tsx
import { createFileRoute } from '@tanstack/react-router'
import { useEffect, useState } from 'react'

export const Route = createFileRoute('/factions')({
  component: FactionsPage,
})

interface Faction {
  id: number
  name: string
  display_name: string
  description: string
  category?: { id: number; name: string; display_name: string }
  required_tags?: string[]
}

function FactionsPage() {
  const [factions, setFactions] = useState<Faction[]>([])
  const [loading, setLoading] = useState(true)
  // ... create/edit modal state

    })
      .then(r => r.json())

  return (

        <h2>Factions</h2>
        <button onClick={() => { /* open create modal */ }}>Create Faction</button>
      </div>
      {loading ? <div>Loading...</div> : (
        <table className="data-table">
          <thead>
            <tr>
              <th>Name</th>
              <th>Display Name</th>
              <th>Category</th>
              <th>Description</th>
              <th>Required Tags</th>
            </tr>
          </thead>
          <tbody>
            {factions.map(f => (
              <tr key={f.id}>
                <td>{f.name}</td>
                <td>{f.display_name}</td>
                <td>{f.category?.display_name || '—'}</td>
                <td>{f.description}</td>
                <td>{f.required_tags?.join(', ') || '—'}</td>
              </tr>
            ))}
          </tbody>
        </table>
      )}
    </div>
  )
}
```
